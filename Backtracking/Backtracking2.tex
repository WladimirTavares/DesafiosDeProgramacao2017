\section{Backtracking}

\begin{frame}{O que é Backtracking?}

\begin{itemize}
\item Backtracking é um algoritmo genérico para encontrar todas ( ou algumas ) soluções para algum problema computacional.
\item O algoritmo constrói incrementalmente possíveis candidatos para a solução e abandona candidatos que não podem gerar soluções para o problema.
\item É um refinamento do método de enumeração bruta.
\item É uma importante ferramenta para resolver problemas de satisfação de restrições.
\end{itemize}

\end{frame}


\begin{frame}{Algoritmo Genérico}

\begin{algorithm}[H]
\caption{Algoritmo Genérico de Backtracking}
\begin{algorithmic}[1]

\Function {solve} {$c$}
\If{completo(c)}
  \If{ aceita(c) }
  \State imprima(c)
  \State \Return true
  \Else
  \State \Return false
  \EndIf
\EndIf

\For{ cada filho $c'$ de $c$ }
    \If{solve(c')}
    \State \Return true
    \EndIf
\EndFor

\EndFunction

\end{algorithmic}

\end{algorithm}

\end{frame}

\begin{frame}[fragile]{Problema 1: Soma de subconjuntos}

\begin{lstlisting}[language=C]
int solve(vector <int> & objetos, int meta, int i, int soma){
    if( i == (int)objetos.size() ){
        if( soma == meta ) return 1;
        else return 0;
    }
    if( solve(objetos, meta, i+1, soma) )
        return true;
    if( solve(objetos, meta, i+1, soma + objetos[i]) )
        return true;
    return false;
}

\end{lstlisting}
}

\subsection{Solução baseada em vetores}

\begin{frame}{Problema 2: N-Queen}

Variáveis booleanas:
\begin{itemize}
  \item diagonais principais:  $d1[-(N-1) \ldots N-1]$
  \item diagonais secundarias: $d2[0 \ldots 2 \times (n-1)]$
  \item linha: $linha [0 \ldots n]$
\end{itemize}

\end{frame}

\begin{frame}{Problema 2: N-Queen}

\begin{algorithm}[H]
\caption{N-Queen}
\begin{algorithmic}[1]
\Function{queen}{j}
\If{Se solução completa válida}
\State registre solução
\EndIf

\For{ cada i tal que $linha[i] == true$ }
\If{ $d1[i-j] == true$ E $d2[i+2] == true$}
\State $d1[i-j] \leftarrow false$
\State $d2[i+j] \leftarrow false$
\State $linha[i] \leftarrow false$
\State queen(j+1)
\State $d1[i-j] \leftarrow true$
\State $d2[i+j] \leftarrow true$
\State $linha[i]\leftarrow true$

\EndIf
\EndFor


\EndFunction
\end{algorithmic}
\end{algorithm}

\end{frame}

void queen(int j, int n){
  if( j == n ){
        contador++;
    }else{
        for(int i = 0; i < n; i++){
            if(  linha[i] && d1[i-j+n-1] && d2[i+j]){
                linha[i] = false;
                d1[i-j+n-1] = false;
                d2[i+j] = false;
                x[j] = i;
                queen(j+1, n);
                linha[i] = true;
                d1[i-j+n-1] = true;
                d2[i+j] = true;
            }
        }
    }
}

\section{Problema 3 - Resta um}

\begin{frame}{Problema 3 - Resta um}

\begin{itemize}

\item[P3] Resta um é jogo interessante. Neste jogo, você tem um tabuleiro com dozes buracos em
uma linha. No começo de cada jogo, alguns buracos estão ocupados por pedrinhas. Um movimento é possível se há uma linha reta de três buracos adjacentes, chamados A, B e C, onde A e B estão com uma pedrinha e C está vazio. O movimento consiste em mover a pedrinha de A para C e retirar a pedrinha que está em B. Sua missão é encontrar uma sequência de movimentos tal que reste o menor número de pedrinhas no tabuleiro.

\end{itemize}
\end{frame}

\section{Problema 4: Passeio do Cavalo}

\begin{frame}{Problema 4: Passeio do Cavalo}

\begin{itemize}
  \item[P4] Faça um cavalo percorrer todas as casas de um tabuleiro de xadrez de forma a não repetir nenhuma posição pela qual já passou começando da posição (0,0).
\end{itemize}
\end{frame}

\begin{frame}{Problema 4: Passeio do cavalo}

\begin{algorithm}[H]
\caption{Passeio do cavalo}
\begin{algorithmic}[1]
\Function{cavalo}{}
\If{tabuleiro  está completo }
\State imprima tabuleiro
\State \Return sucesso
\EndIf

\While{ existe movimento não realizado }
\State selecione um movimento
\If{ movimento aceitável}
\State registre movimento
\State chamada recursiva cavalo
\If { chamada não sucedida }
\State desfaz movimento
\Else 
\State \Return sucesso
\EndIf
\EndIf
\EndWhile

\State \Return fracasso
\EndFunction
\end{algorithmic}
\end{algorithm}

\end{frame}

\begin{frame}{Problema 5: Flood Fill}

\begin{algorithm}[H]
\caption{FloodFill}
\begin{algorithmic}[1]
\Function{construções}{}
\State $cont \leftarrow 0$
\For{cada construção na posição (i,j) }
\State apaga(i,j)
\State $cont \leftarrow cont + 1$
\EndFor
\Return cont
\EndFunction
\Function{apaga}{int i, int j}
\State $imagem[i][j] \leftarrow 0$
\For{cada $(u,v) \leftarrow adjacente(i,j)$ }
\If{ $(u,v)$ está no tabuleiro }
\If{ $imagem[u][v] \neq 0$}
\State apaga(u,v)
\EndIf
\EndIf
\EndFor
\EndFunction
\end{algorithmic}
\end{algorithm}

\end{frame}
