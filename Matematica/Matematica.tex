\begin{frame}{Exponenciação Rápida}

\begin{itemize}
  \item Computação recursiva de $a^n$

\begin{center}
$a^n =
\begin{cases}
1  &, n = 0\\
a  &, n = 1\\
(a^{n/2}) &, n \mbox{ é par}\\
a(a^{n-1/2})2 &, n \mbox{ é ímpar}\\
\end{cases}
$
\end{center}

\end{itemize}

\end{frame}

\begin{frame}[fragile]{Implementação recursiva}
\Fontvi{
\begin{lstlisting}[language=C]
double pow(double a, int n){
  if(n==0) return 1;
  if(n==1) return a;
  double t = pow(a,n/2);
  return t*t*pow(a,n%2);
}
\end{lstlisting}
}

\begin{itemize}
  \item Complexidade de tempo : $O(log n)$
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Eliminação de Gauss}
\Fontvi{
\begin{lstlisting}[language=C]
LD MATRIX::Gauss()
{
  int i,j,k;
  LD det=1.0, r;
  for(i=0;i<n;i++)
  {
   for(j=i, k=-1; j<n; j++) if(fabs(a[j][i])>EPS)
   { k=j; j=n+1; }
   if(k<0) {
     n=0; return 0.0;
   }
   if(k != i) { swap(a[i], a[k]); det=-det; }
   r=a[i][i]; det*=r;
   for(j=i; j<m; j++) a[i][j]/=r;
   for(j=i+1; j<n; j++)
   {
    r=a[j][i];
    for(k=i; k<m; k++) a[j][k]-=a[i][k]*r;
   }
  }


\end{lstlisting}
}

int MATRIX::inverse()
 // assume n=m. returns 0 if not invertible
 {
  int i, j, ii;
  MATRIX T; T.resize(n, 2*n);
  for(i=0;i<n;i++) for(j=0;j<n;j++) T.a[i][j]=a[i][j];
  for(i=0;i<n;i++) T.a[i][i+n]=1.0;
  T.Gauss();
  if(T.n==0) return 0;
  for(i=0;i<n;i++) for(j=0;j<n;j++) a[i][j]=T.a[i][j+n];
  return 1;
}

\end{lstlisting}

typedef map<int, int> prime_map;
void squeeze(prime_map& M, int& n, int p) {
  for (; n % p == 0; n /= p) M[p]++;
}
prime_map factor(int n) {
  prime_map M;
  if (n < 0) return factor(-n);
  if (n < 2) return M;
  squeeze(M, n, 2);
  squeeze(M, n, 3);
  int maxP = sqrt(n) + 2;
  for (int p = 5; p < maxP; p += 6) {

//Complexity: O(n^2/(log n)^2)
vector <int> SieveTrialFactorization(int n){
  vector <int> primes;
  primes.push_back(2);
  primes.push_back(3);
  for(int p = 5; p <= n; p += 2)
  {
    bool is_prime = true;
    for(int i = 1; i < primes.size(); i++)
    {
      int k = primes[i];
      if(k*k > p) break;
      if( p%k == 0)
      {
        is_prime = false;
        break;
      }
    }
    if( is_prime)
    {
      primes.push_back(p);
    }
  }
  return primes;
}

\end{lstlisting}
}

\begin{frame}[fragile]{Crivo de Erathostenes}

\Fontvi{
\begin{lstlisting}[language=C]
//Complexity: O(n log log n)
vector<int> SieveErathostenes(int n){
  vector <bool> S;
  vector <int> primes;

  for(int p = 0; p <= n; p++) S.push_back(true);

  S[1] = false;
  for(int p = 2; p*p <= n; p++){
    if(S[p]){
      for(int q = p*p; q <= n; q += p)
      {
        S[q] = false;
      }
    }
  }
  for(int p = 2; p <= n; p++){
    if(S[p]) primes.push_back(p);
  }
  return primes;
}

\end{lstlisting}
}


\end{frame}

\begin{frame}[fragile]{Crivo Linear de Pritchard}

\Fontvi{
\begin{lstlisting}[language=C]
//Complexity: O(n)
vector <int> LinearSievePritchard(int n){
  vector <bool> S;
  for(int p = 0; p <= n; p++) S.push_back(true);
  S[1] = false;
  vector <int> primes = SieveErathostenes(ceil(sqrt(n)));

  for(int f = 2; f <= n/2; f=f+1){
    for(int i = 0; i < primes.size(); i++)
    {
      int p = primes[i];
      if(p > n/f) break;
      S[p*f] = false;
      if(f%p==0) break;
    }
  }
  for(int k = primes[primes.size()-1] + 2; k <= n; k += 2)
  {
    if(S[k]) primes.push_back(k);
  }
  return primes;
}

\end{lstlisting}
}

\begin{frame}{Máximo Divisor Comum}
\begin{itemize}
\item $gcd(a,b)$ : maior inteiro que divide $a$ e $b$.
\item $gcd(a,b) = gcd(a, b-a)$
\item $gcd(a,b) = gcd(b, a\%b)$
\item $gcd(a,0) = a$
\item $gcd(a,b)$ é o menor inteiro positivo em $\{ax + by | x,y \in \mathbb{Z}\}$
\end{itemize}
\end{frame}

\begin{frame}{Algoritmo de Euclides}

\begin{itemize}
\item Example:

\begin{center}
\begin{tabular}{lll}
gcd(15,12) & = & gcd(12,15\%12)\\
           & = & gcd(12, 3)\\
           & = & gcd(3, 12 \%3)\\
           & = & gcd(3, 0)\\
           & = & 3\\
\end{tabular}
\end{center}
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Implementação}

\Fontvi{
\begin{lstlisting}[language=C]
int gcd(int a, int b){
    while(b){
        int r = a%b;
        a = b;
        b = r;
    }
    return a;
}
\end{lstlisting}
}

\begin{itemize}
  \item Complexidade de tempo: O( log (a+b) )
\end{itemize}

\end{frame}

\begin{frame}{Algoritmo de Euclides Estendido}

\begin{itemize}
  \item $gcd(a,b) = ax + by$ para algum inteiro $x$ e $y$
  \item Os inteiros $x$ e $y$ pode ser encontrados usando o algoritmo de euclides estendido.
  \item Example: $a = 15, b = 12$

  \begin{center}
  \begin{tabular}{cccc}
  r  &  q & x & y\\
  \hline
  7 & * & 1  & 0 \\
  5 & * & 0  & 1 \\
  2 & 1 & 1  & -1\\
  1 & 2 & -2 & 3\\
  \end{tabular}
  \end{center}

  \item gcd(7,5) = 7(-2)+5(3) = 1

\end{itemize}

\end{frame}

\begin{frame}[fragile]{Algoritmo de Euclides Estendido}

\Fontvi{
\begin{lstlisting}[language=C]
int gcd(int a, int b, int & x, int & y)
{
  int q, t;
  int x1, y1;
  x = 1, y = 0;
  x1 = 0, y1 = 1;
  while(b)
  {
    q = a/b;
    t = b;
    b = a - q*b;
    a = t;
    t = x1;
    x1 = x - q*x1;
    x = t;
    t = y1;
    y1 = y - q*y1;
    y = t;
  }
  return a;
}

\end{lstlisting}
}


\end{frame}

\begin{frame}{Inverso Multiplicativo Modular}

\begin{itemize}
\item $a^{-1}$ é o inverso multiplicativo de a módulo n se somente se $aa^{-1} = 1 (mod~ n)$
\item Se $gcd(a,n) = 1$ então $ax + ny = 1$ para algum $x,y \in \mathbb{Z}$
\item Tomando módulo $n$, temos $ax = 1 (mod~ n)$
\item $x$ é o inverso de $a (mod~ n)$.
\item Example: Qual é o inverso de $55^{-1} (mod~ 89)$?
\begin{itemize}
  \item gcd(55,89) = 1 = 55(34) + 89(-21)
  \item 55(34) = 1 (mod 89)
  \item 34 é inverso de 55 módulo 89.
\end{itemize}
\end{itemize}

\end{frame}

typedef long long int lli;
lli binom2(int n, int k){
    if(k > n) return  0;
    if(n == k) return 1;
    if(k > n-k) k = n-k;
    lli c = 1;
    for(int i = 1; i <= k; i++){
        c *= n--;
        c /= i;
    }
    return c;

\begin{frame}[fragile]{Computando Coeficiente Binomial}

\begin{itemize}
\item Use o triângulo de Pascal
\begin{center}
$\displaystyle \binom{n}{k} = \begin{cases}
1 &, k = 0\\
1 &, k = n \\
\binom{n-1}{k-1} + \binom{n-1}{k} &, \mbox{caso contrário}\\
\end{cases}$

\end{center}
\end{itemize}

\begin{frame}{Sequência de Fibonacci}


\begin{itemize}
\item Matriz de Recorrência
\begin{equation}
\begin{bmatrix}
F_{n+1} \\
F_{n}
\end{bmatrix}
=
\begin{bmatrix}
1 & 1 \\
1 & 0 \\
\end{bmatrix}
\times
\begin{bmatrix}
F_{n} \\
F_{n-1}
\end{bmatrix}=
\begin{bmatrix}
1 & 1 \\
1 & 0 \\
\end{bmatrix}^n
\times
\begin{bmatrix}
F_{1} \\
F_{0}
\end{bmatrix}
\end{equation}
\item Pode ser calculado em $O(lg n)$ usando exponenciação rápida
\end{itemize}
